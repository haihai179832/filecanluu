    <script>
        const darkScreen = document.querySelector('.dark-screen')
        const openNav = document.querySelector('#content .open-nav')
        const navBars = document.querySelector('#navbars')
        const closeNav = document.querySelector('#navbars .close-nav')
        openNav.addEventListener('click', () => {
            navBars.style.display = 'block'
            darkScreen.style.display= 'block'
        })
        closeNav.addEventListener('click', () => {
            navBars.style.display = 'none'
            darkScreen.style.display= 'none'
        })
        darkScreen.addEventListener('click', () => {
            navBars.style.display = 'none'
            darkScreen.style.display= 'none'
        })

        document.addEventListener('DOMContentLoaded', function() {
    const header = document.querySelector('.header');
    const content = document.querySelector('#content .content');
    let lastScrollTop = 0;

    content.addEventListener('scroll', function() {
        let currentScrollTop = content.scrollTop;

        if (currentScrollTop > lastScrollTop) {
            // Cuộn xuống
            header.classList.add('hidden');
        } else {
            // Cuộn lên
            header.classList.remove('hidden');
        }
        
        lastScrollTop = currentScrollTop <= 0 ? 0 : currentScrollTop; // Đảm bảo lastScrollTop không âm
    });
});
